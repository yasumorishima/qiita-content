# データ取得

スクレピング。サイトに迷惑にならないように。

<pre><code class="language-python">
pip install requests beautifulsoup4 pandas

import requests
from bs4 import BeautifulSoup
import csv
from datetime import datetime

place_codeA = [44, 46, 43, 82, 91, 14, 62, 52, 34]
place_codeB = [47662, 47670, 47626, 47807, 47936, 47412, 47772, 47590]
place_name = ["東京", "横浜", "熊谷", "福岡", "沖縄", "札幌", "大阪", "仙台"]

base_url = "http://www.data.jma.go.jp/obd/stats/etrn/view/daily_s1.php?prec_no=%s&block_no=%s&year=%s&month=%s&day=1&view=p1"

def str2float(str):
    try:
        return float(str)
    except:
        return 0.0

if __name__ == "__main__":
    current_year = datetime.now().year
    current_month = datetime.now().month
    for place in place_name:
        All_list = [['年月日', '降水量', '気温_平均', '気温_最高', '気温_最低', '湿度_平均', '湿度_最小', '日照時間']]
        print(place)
        index = place_name.index(place)
        for year in range(2020, current_year + 1):
            print(year)
            end_month = 13 if year != current_year else current_month + 1
            for month in range(1, end_month):
                r = requests.get(base_url%(place_codeA[index], place_codeB[index], year, month))
                r.encoding = r.apparent_encoding
                soup = BeautifulSoup(r.text)
                rows = soup.findAll('tr',class_='mtx')
                rows = rows[4:]
                for row in rows:
                    data = row.findAll('td')
                    rowData = []
                    rowData.append(str(year) + "/" + str(month).zfill(2) + "/" + str(data[0].string).zfill(2))
                    rowData.append(str2float(data[3].string))
                    rowData.append(str2float(data[6].string))
                    rowData.append(str2float(data[7].string))
                    rowData.append(str2float(data[8].string))
                    rowData.append(str2float(data[9].string))
                    rowData.append(str2float(data[10].string))
                    rowData.append(str2float(data[16].string))
                    All_list.append(rowData)
        with open(place + '.csv', 'w',encoding="utf_8_sig") as file:
            writer = csv.writer(file, lineterminator='\n')
            writer.writerows(All_list)

place_name_jp = ["東京", "横浜", "熊谷", "福岡", "沖縄", "札幌", "大阪",  "仙台"]
place_name_en = ["Tokyo", "Yokohama", "Kumagaya", "Fukuoka", "Okinawa", "Sapporo", "Osaka", "Nagoya", "Sendai"]
place_dict = dict(zip(place_name_jp, place_name_en))

# 各都市のデータを読み込み、一つのDataFrameに結合
df_list = []
for place_jp, place_en in place_dict.items():
    df = pd.read_csv(f'{place_jp}.csv')
    df['City'] = place_en
    df_list.append(df)

combined_df = pd.concat(df_list)

# 結果をCSVファイルに保存
combined_df.to_csv('combined_data.csv', index=False)

import seaborn as sns
import pandas as pd
import matplotlib.pyplot as plt

# 都市名と英語名の対応
place_name = ["東京", "横浜", "熊谷", "福岡", "沖縄", "札幌", "大阪"]
place_name_english = ["Tokyo", "Yokohama", "Kumagaya", "Fukuoka", "Okinawa", "Sapporo", "Osaka"]

for i, city in enumerate(place_name):
    df = pd.read_csv(f'{city}.csv')
    df['年月日'] = pd.to_datetime(df['年月日'])
    df.set_index('年月日', inplace=True)
    df = df[(df.index.month == 6) | (df.index.month == 7)]  # 6月と7月のデータだけを抽出
    df = df[df['気温_最高'] != 0]  # 最高気温が0のデータを除外
    df['Day of Year'] = df.index.dayofyear
    df['Year'] = df.index.year
    df_pivot = df.pivot(index='Year', columns='Day of Year', values='気温_最高')
    
    # df_pivotが空でないことを確認
    if not df_pivot.empty:
        plt.figure(figsize=(10, 8))
        sns.heatmap(df_pivot, cmap='coolwarm', cbar_kws={'label': 'Highest Temperature'})
        plt.title(place_name_english[i])
        plt.xlabel('Date (6/1 - 7/31)')
        plt.ylabel('Year')
        plt.show()
    else:
        print(f"No data available for {city}. Skipping...")
