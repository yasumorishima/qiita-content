*   ST - slutter / slider / cutter / sweeper　※実際の投球を見るとスイーパー
*   FS - splitter
*   FF - 4-seam fastball
*   CU - curveball
*   SL - slider
*   FC - cutter
*   SI - sinker (2-seam)

<pre><code>import pandas as pd

def pitch_counts(df):
    # 左打者と右打者に対する投球データを抽出
    df_L = df[df['stand'] == 'L']
    df_R = df[df['stand'] == 'R']

    # 各カテゴリーでの球種の出現回数をカウント
    total_counts = df['pitch_type'].value_counts()
    left_counts = df_L['pitch_type'].value_counts()
    right_counts = df_R['pitch_type'].value_counts()

    # 出現回数をデータフレームにまとめる
    pitch_counts_table = pd.DataFrame({'Total': total_counts, 'Left Batter': left_counts, 'Right Batter': right_counts})

    # NaNを0に置き換える
    pitch_counts_table.fillna(0, inplace=True)

    # カウントを整数に変換する
    pitch_counts_table = pitch_counts_table.astype(int)

    return pitch_counts_table

# 続けて、球種カウントのコードを実行します。
# Split the data by year
df_2021 = df_660271_all_dates[df_660271_all_dates['game_year'] == 2021]
df_2022= df_660271_all_dates[df_660271_all_dates['game_year'] == 2022]
df_2023 = df_660271_all_dates[df_660271_all_dates['game_year'] == 2023]

# Get pitch counts for each year
pitch_counts_2021 = pitch_counts(df_2021)
pitch_counts_2022 = pitch_counts(df_2022)
pitch_counts_2023 = pitch_counts(df_2023)

# Display the results
print("2021年の球種カウント:")

print("\n2022年の球種カウント:")

import pandas as pd

def plot_spin_rate_by_year_boxplot(df_2021, df_2022, df_2023, pitch_types):
    pitch_types = [pt for pt in pitch_types if pt is not None and pd.notna(pt)]  # nan を除外
    fig, axs = plt.subplots(len(pitch_types), 1, figsize=(8, len(pitch_types) * 4))

    for i, pitch_type in enumerate(pitch_types):
        ax = axs[i]

        grouped_2021 = df_2021[df_2021['pitch_type'] == pitch_type].dropna(subset=['release_spin_rate'])
        grouped_2022 = df_2022[df_2022['pitch_type'] == pitch_type].dropna(subset=['release_spin_rate'])
        grouped_2023 = df_2023[df_2023['pitch_type'] == pitch_type].dropna(subset=['release_spin_rate'])

        data_to_plot = []
        labels = []
        if not grouped_2021.empty:
            data_to_plot.append(grouped_2021['release_spin_rate'])
            labels.append('2021')
        if not grouped_2022.empty:
            data_to_plot.append(grouped_2022['release_spin_rate'])
            labels.append('2022')
        if not grouped_2023.empty:
            data_to_plot.append(grouped_2023['release_spin_rate'])
            labels.append('2023')

        if data_to_plot:
            bp = ax.boxplot(data_to_plot, labels=labels)
            for j, d in enumerate(data_to_plot):
                mean_val = d.mean()
                ax.text(j + 0.8, mean_val, f"{mean_val:.1f}", ha='center', va='bottom', fontsize=10, color='blue')

        ax.set_title(f"Pitch Type: {pitch_type}")
        ax.set_xlabel('Year')
        ax.set_ylabel('Release Spin Rate')

        # 罫線
        ax.grid(which='both', linestyle='--', color='gray', alpha=0.5)
