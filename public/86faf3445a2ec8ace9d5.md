<pre><code>import pandas as pd

def pitch_counts(df):
    # 左打者と右打者に対する投球データを抽出
    df_L = df[df['stand'] == 'L']
    df_R = df[df['stand'] == 'R']

    # 各カテゴリーでの球種の出現回数をカウント
    total_counts = df['pitch_type'].value_counts()
    left_counts = df_L['pitch_type'].value_counts()
    right_counts = df_R['pitch_type'].value_counts()

    # 出現回数をデータフレームにまとめる
    pitch_counts_table = pd.DataFrame({'Total': total_counts, 'Left Batter': left_counts, 'Right Batter': right_counts})

    # NaNを0に置き換える
    pitch_counts_table.fillna(0, inplace=True)

    # カウントを整数に変換する
    pitch_counts_table = pitch_counts_table.astype(int)

    return pitch_counts_table

# 2021年と2023年のデータに対して球種カウントを取得
pitch_counts_2021 = pitch_counts(df_628317_2021_single_day)
pitch_counts_2023 = pitch_counts(df_628317_2023_single_day)

# 結果を表示
print("2021年の球種カウント:")
print(pitch_counts_2021)
print("\n2023年の球種カウント:")
print(pitch_counts_2023)

def plot_pitch_distribution(df, year):

    axs[1].set_ylabel('')

    # vs Right batter
    df_R['pitch_type'].value_counts().plot(kind='pie', ax=axs[2], autopct='%.1f%%', colors=[colors[key] for key in df_R['pitch_type'].value_counts().index])
    axs[2].set_title('vs Right batter')
    axs[2].set_ylabel('')

def plot_release_point_by_year(df_2021, df_2023):
    # 両方のデータフレームを結合
    combined_df = pd.concat([df_2021, df_2023])

    # データを pitch_type ごとにグループ分けする
    grouped = combined_df.groupby('pitch_type')

    # グラフの数と行列を指定
    nrows = 2
    ncols = 3

    titles = []

        titles.append(pitch_type)

        # サブプロットに対応する行と列のインデックスを計算
        row = idx // ncols
        col = idx % ncols

        ax = axes[row][col]

        # 2021年と2023年のデータをそれぞれプロット
        data_2021 = data[data.index.isin(df_2021.index)]
        data_2023 = data[data.index.isin(df_2023.index)]

# リリーススピンレートを球種ごと (Boxplot)

このグラフでは、各球種について、2021年と2023年のリリーススピンレートをボックスプロットで表示しています。

ボックスプロットは、データのばらつきや分布を視覚的に把握するために使用されるグラフです。箱の上下の線は第1四分位数（Q1）と第3四分位数（Q3）を示し、箱の中央の線は中央値（Q2）を示しています。また、外れ値を示す点が含まれている場合もあります。

<pre><code>import matplotlib.pyplot as plt
import pandas as pd

def plot_spin_rate_by_year_boxplot(df_2021, df_2023, pitch_types):
    fig, axs = plt.subplots(len(pitch_types), 1, figsize=(8, len(pitch_types) * 4))

    for i, pitch_type in enumerate(pitch_types):
        ax = axs[i]

        grouped_2021 = df_2021[df_2021['pitch_type'] == pitch_type].dropna(subset=['release_spin_rate'])
        grouped_2023 = df_2023[df_2023['pitch_type'] == pitch_type].dropna(subset=['release_spin_rate'])

        data_to_plot = []
        labels = []
        if not grouped_2021.empty:
            data_to_plot.append(grouped_2021['release_spin_rate'])
            labels.append('2021')
        if not grouped_2023.empty:
            data_to_plot.append(grouped_2023['release_spin_rate'])
            labels.append('2023')

        if data_to_plot:
            bp = ax.boxplot(data_to_plot, labels=labels)
            for j, d in enumerate(data_to_plot):
                mean_val = d.mean()
                ax.text(j + 0.8, mean_val, f"{mean_val:.1f}", ha='center', va='bottom', fontsize=10, color='blue')

        ax.set_title(f"Pitch Type: {pitch_type}")
        ax.set_xlabel('Year')
        ax.set_ylabel('Release Spin Rate')

        # 罫線
        ax.grid(which='both', linestyle='--', color='gray', alpha=0.5)

    fig.suptitle('Release Spin Rate by Year and Pitch Type', fontsize=16, y=1.02)
    plt.tight_layout(pad=3)
    plt.show()

# 2021年と2023年のデータに対してリリーススピンレートをプロット
pitch_types_2021 = df_628317_2021_single_day['pitch_type'].unique()
pitch_types_2023 = df_628317_2023_single_day['pitch_type'].unique()

# 両年に存在する球種と、2021年には存在しないが2023年に存在する球種を取得
all_pitch_types = set(pitch_types_2021) | set(pitch_types_2023)

plot_spin_rate_by_year_boxplot(df_628317_2021_single_day, df_628317_2023_single_day, all_pitch_types)
