# 菊池雄星投手のピッチングデータを取得

このコードは、菊池雄星（プレイヤーID: 579328）のピッチングデータを取得し、一つのデータフレームにまとめるものです。Pythonのライブラリpybaseballを使って、指定された日付のピッチングデータを取得しています。

<pre><code class>!pip install pybaseball

    '2022-08-08', '2022-08-15', '2022-08-18', '2022-08-23', '2022-08-26', '2022-08-28',
    '2022-09-03', '2022-09-06', '2022-09-11', '2022-09-16', '2022-09-25', '2022-09-28',

# 左打者と右打者に対する球種の使用球数

このコードでは、菊池雄星の投球データから、各年ごとに左打者と右打者に対する球種の使用回数を集計し、表示しています。

<pre><code class>import pandas as pd

def pitch_counts(df):
    # 左打者と右打者に対する投球データを抽出
    df_L = df[df['stand'] == 'L']
    df_R = df[df['stand'] == 'R']

    # 各カテゴリーでの球種の出現回数をカウント
    total_counts = df['pitch_type'].value_counts()
    left_counts = df_L['pitch_type'].value_counts()
    right_counts = df_R['pitch_type'].value_counts()

    # 出現回数をデータフレームにまとめる
    pitch_counts_table = pd.DataFrame({'Total': total_counts, 'Left Batter': left_counts, 'Right Batter': right_counts})

    # NaNを0に置き換える
    pitch_counts_table.fillna(0, inplace=True)

    # カウントを整数に変換する
    pitch_counts_table = pitch_counts_table.astype(int)

    return pitch_counts_table

# 続けて、球種カウントのコードを実行します。
# Split the data by year
df_2022= df_579328_all_dates[df_579328_all_dates['game_year'] == 2022]
df_2023 = df_579328_all_dates[df_579328_all_dates['game_year'] == 2023]

# Get pitch counts for each year
pitch_counts_2022 = pitch_counts(df_2022)
pitch_counts_2023 = pitch_counts(df_2023)

# Display the results
print("\n2022年の球種カウント:")
print(pitch_counts_2022)
print("\n2023年の球種カウント:")
print(pitch_counts_2023)

# 球種別の投球コースを散布図で表示

<pre><code class>import matplotlib.pyplot as plt

def plot_pitch_location(df, year):
    # データを pitch_type ごとにグループ分けする
    grouped = df.groupby('pitch_type')

    colors = {'SL': 'purple', 'FF': 'blue', 'CH': 'green', 'CU': 'orange', 'FC': 'red'}

    # pitch_type ごとに、'plate_x' を X 軸、'plate_z' を Y 軸とした散布図を作成する
    for pitch_type, data in grouped:
        plt.scatter(data['plate_x'], data['plate_z'], label=pitch_type, color=colors[pitch_type])

    # ストライクゾーン
    x = [-0.88, 0.88, 0.88, -0.88, -0.88]
    y = [1.51, 1.51, 3.4, 3.4, 1.51]
    plt.plot(x, y, linestyle='--', color='red')

    # 凡例を表示する
    plt.legend(bbox_to_anchor=(1.05, 1), loc='upper left', borderaxespad=0)

    plt.xlim(-3, 3)
    plt.ylim(-2, 6)

    plt.xlabel('Plate X')
    plt.ylabel('Plate Z')

    # 罫線
    plt.grid(which='both', linestyle='--', color='gray', alpha=0.5)

    plt.title(f'Pitch Location in {year}')
    
    # グラフを表示する
    plt.show()

# 各年の球種別の投球コース並べて表示

このコードは、2022年と2023年の菊池雄星の投球データをもとに、各球種ごとに投球位置をプロットして比較できるように改良されたものです。plot_pitch_location関数が引数として2つのデータフレームと対応する年を取り、各年の球種別の投球位置を並べて表示します。

<pre><code class>import matplotlib.pyplot as plt

def plot_pitch_location(df1, df2, year1, year2):
    # データを pitch_type ごとにグループ分けする
    grouped1 = df1.groupby('pitch_type')
    grouped2 = df2.groupby('pitch_type')

    colors = {'SL': 'purple', 'FF': 'blue', 'CH': 'green', 'CU': 'orange', 'FC': 'red'}

    pitch_types = set(df1['pitch_type'].unique()).union(set(df2['pitch_type'].unique()))
    num_pitch_types = len(pitch_types)
    
    fig, axs = plt.subplots(num_pitch_types, 2, figsize=(10, 4 * num_pitch_types), constrained_layout=True)
    
    # ストライクゾーンの座標
    x = [-0.88, 0.88, 0.88, -0.88, -0.88]
    y = [1.51, 1.51, 3.4, 3.4, 1.51]
    
    for i, pitch_type in enumerate(pitch_types):
        data1 = grouped1.get_group(pitch_type) if pitch_type in grouped1.groups else None
        data2 = grouped2.get_group(pitch_type) if pitch_type in grouped2.groups else None

        # 2022
        if data1 is not None:
            axs[i, 0].scatter(data1['plate_x'], data1['plate_z'], label=pitch_type, color=colors[pitch_type])
            axs[i, 0].fill(x, y, color='r', alpha=0.3)
        axs[i, 0].set_xlim(-3, 3)
        axs[i, 0].set_ylim(-2, 6)
        axs[i, 0].set_xlabel('Plate X')
        axs[i, 0].set_ylabel('Plate Z')
        axs[i, 0].grid(which='both', linestyle='--', color='gray', alpha=0.5)
        axs[i, 0].set_title(f'{pitch_type} Pitch Location in {year1}')

        # 2023
        if data2 is not None:
            axs[i, 1].scatter(data2['plate_x'], data2['plate_z'], label=pitch_type, color=colors[pitch_type])
            axs[i, 1].fill(x, y, color='r', alpha=0.3)
        axs[i, 1].set_xlim(-3, 3)
        axs[i, 1].set_ylim(-2, 6)
        axs[i, 1].set_xlabel('Plate X')
        axs[i, 1].set_ylabel('Plate Z')
        axs[i, 1].grid(which='both', linestyle='--', color='gray', alpha=0.5)
        axs[i, 1].set_title(f'{pitch_type} Pitch Location in {year2}')

    # グラフを表示する
    plt.show()

# 2022年と2023年のデータに対して投球位置をプロット
plot_pitch_location(df_2022, df_2023, 2022, 2023)

# 投球コースのヒートマップをプロットして比較3

このコードは、2022年と2023年の各ピッチタイプ（スライダー、フォーシーム・ファストボール、チェンジアップ、カーブ、カット・ファストボール）の投球位置のヒートマップを作成し、ストライクゾーンを表示します。ヒートマップは、投球位置の密度を色で表現し、ストライクゾーンは黒い破線で描かれます。これにより、各年の投球位置の分布やストライクゾーン内外での投球傾向を視覚的に把握することができます。

さらに、このコードでは、最大および最小座標を計算して、ヒートマップの表示範囲を調整しています。これにより、すべてのヒートマップが同じ座標範囲で表示され、比較が容易になります。

実行して表示されるヒートマップから、2022年と2023年の各ピッチタイプの投球位置の変化や傾向を比較することができます。投球位置がストライクゾーン内にどれだけ集まっているかや、外角・内角での投球の違いなどを観察することができます。

<pre><code class>import seaborn as sns

def plot_pitch_location(df1, df2, year1, year2, pitch_type, x_min, x_max, y_min, y_max):

    plt.subplots_adjust(wspace=0.1, hspace=0.1)

    x = [-0.88, 0.88, 0.88, -0.88, -0.88]

# 投球コースのヒートマップをプロットして比較4

このコードは、2つのデータフレーム（df\_2022 と df\_2023）に基づいて、2つの異なる年（year1 と year2）のピッチタイプ（SL, FF, CH, CU, FC）ごとにピッチ位置のヒートマップをプロットし、ストライクゾーンを追加して表示する関数 plot_pitch_location を定義しています。

<pre><code class>import seaborn as sns

def plot_pitch_location(df1, df2, year1, year2):
    pitch_types = ['SL', 'FF', 'CH', 'CU', 'FC']
    fig, axs = plt.subplots(2, len(pitch_types), figsize=(20, 10), sharex=True, sharey=True)
    plt.subplots_adjust(wspace=0.1, hspace=0.1)

    for i, pitch_type in enumerate(pitch_types):
